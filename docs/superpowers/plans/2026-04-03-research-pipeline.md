```typescript
import { mutation, internalMutation, query } from "./_generated/server"
import { internal } from "./_generated/api"
import { v } from "convex/values"

/**
 * Get the placeholder block for a session (at most 1 exists).
 */
export const getPlaceholderBlock = query({
  args: { sessionId: v.id("sessions") },
  handler: async (ctx, args) => {
    return await ctx.db
      .query("blocks")
      .withIndex("by_session", (q) => q.eq("sessionId", args.sessionId))
      .filter((q) => q.eq(q.field("type"), "placeholder"))
      .first()
  },
})

/**
 * Internal mutation called by runResearchAction on completion.
 * Writes research result into the placeholder block.
 */
export const fillPlaceholderBlock = internalMutation({
  args: {
    blockId: v.id("blocks"),
    content: v.string(),
  },
  handler: async (ctx, args) => {
    await ctx.db.patch(args.blockId, {
      content: args.content,
      placeholderStatus: "filled",
      updatedAt: Date.now(),
    })
  },
})

/**
 * Internal mutation to reset placeholder status (on error).
 */
export const resetPlaceholderBlock = internalMutation({
  args: { blockId: v.id("blocks") },
  handler: async (ctx, args) => {
    await ctx.db.patch(args.blockId, {
      placeholderStatus: "empty",
      updatedAt: Date.now(),
    })
  },
})

/**
 * Start a research generation that fills the session's placeholder block.
 * Enforces: exactly 1 placeholder per session, spec must be present, not already running.
 */
export const startResearch = mutation({
  args: { sessionId: v.id("sessions") },
  handler: async (ctx, args) => {
    // Find placeholder block
    const placeholder = await ctx.db
      .query("blocks")
      .withIndex("by_session", (q) => q.eq("sessionId", args.sessionId))
      .filter((q) => q.eq(q.field("type"), "placeholder"))
      .first()

    if (!placeholder) {
      throw new Error("No placeholder block found for this session")
    }
    if (!placeholder.placeholderSpec?.trim()) {
      throw new Error("Placeholder spec is required before running research")
    }
    if (placeholder.placeholderStatus === "running") {
      throw new Error("Research is already in progress")
    }

    // Create generation record
    const now = Date.now()
    const generationId = await ctx.db.insert("generations", {
      sessionId: args.sessionId,
      provider: "claude",
      status: "streaming",
      text: "",
      createdAt: now,
      updatedAt: now,
    })

    // Mark placeholder as running
    await ctx.db.patch(placeholder._id, {
      placeholderStatus: "running",
      updatedAt: now,
    })

    // Schedule research action
    await ctx.scheduler.runAfter(0, internal.claudeNode.runResearchAction, {
      generationId,
      sessionId: args.sessionId,
      blockId: placeholder._id,
    })

    return { generationId, blockId: placeholder._id }
  },
})
```

```typescript
/**
 * Fill a placeholder block by running web research via Claude with WebSearch+WebFetch.
 * Called by startResearch mutation via scheduler.
 */
export const runResearchAction = action({
  args: {
    generationId: v.id("generations"),
    sessionId: v.id("sessions"),
    blockId: v.id("blocks"),
  },
  handler: async (ctx, args): Promise<void> => {
    const startTime = Date.now()

    // Load placeholder block spec
    const block = await ctx.runQuery(internal.research.getPlaceholderBlock, {
      sessionId: args.sessionId,
    })
    if (!block || !block.placeholderSpec) {
      await ctx.runMutation(internal.generations.fail, {
        generationId: args.generationId,
        error: "Placeholder block or spec not found",
      })
      return
    }

    // Assemble system prompt from PERMANENT blocks
    const blocks = await ctx.runQuery(internal.blocks.listBySessionInternal, {
      sessionId: args.sessionId,
    })
    let systemPrompt = assembleSystemPromptWithContext(blocks, undefined, "brainstorm")
    systemPrompt = (systemPrompt ?? "") + RESEARCH_SUFFIX

    const prompt = block.placeholderSpec

    const trace = createGeneration(
      "claude-research",
      { sessionId: args.sessionId, provider: "claude", model: "claude-code" },
      { systemPrompt, prompt }
    )

    let buffer = ""
    let fullText = ""
    let lastFlush = Date.now()
    const throttleMs = 100
    let inputTokens: number | undefined
    let outputTokens: number | undefined
    let costUsd: number | undefined
    let resolvedModel: string | undefined
    const abortController = new AbortController()
    const stderrChunks: string[] = []

    const isCancelled = async (): Promise<boolean> => {
      const gen = await ctx.runQuery(internal.generations.getInternal, {
        generationId: args.generationId,
      })
      return gen?.status === "cancelled"
    }

    const flushBuffer = async () => {
      if (buffer.length > 0) {
        await ctx.runMutation(internal.generations.appendChunk, {
          generationId: args.generationId,
          chunk: buffer,
        })
        buffer = ""
        lastFlush = Date.now()
      }
    }

    try {
      let hasReceivedStreamEvents = false

      for await (const message of claudeQuery({
        prompt,
        options: {
          abortController,
          allowedTools: ["WebSearch", "WebFetch"],
          maxTurns: 10,
          systemPrompt,
          pathToClaudeCodeExecutable: getClaudeCodePath(),
          includePartialMessages: true,
          maxBudgetUsd: 1.00,
          stderr: (data: string) => { stderrChunks.push(data) },
        },
      })) {
        const msgType = (message as Record<string, unknown>).type as string

        if (msgType === "stream_event") {
          const event = (message as Record<string, unknown>).event as Record<string, unknown> | undefined
          if (event?.type === "content_block_delta") {
            const delta = event.delta as Record<string, unknown> | undefined
            if (delta?.type === "text_delta" && typeof delta.text === "string") {
              hasReceivedStreamEvents = true
              buffer += delta.text
              fullText += delta.text
              const now = Date.now()
              if (now - lastFlush >= throttleMs) {
                await flushBuffer()
                if (await isCancelled()) {
                  abortController.abort()
                  break
                }
              }
            }
          }
        }

        if (msgType === "assistant" && !resolvedModel) {
          const msg = message as Record<string, unknown>
          const msgContent = msg.message as Record<string, unknown> | undefined
          if (typeof msgContent?.model === "string") resolvedModel = msgContent.model
        }

        if (msgType === "assistant" && !hasReceivedStreamEvents) {
          const msg = message as Record<string, unknown>
          const msgContent = msg.message as Record<string, unknown> | undefined
          const content = msgContent?.content as Array<Record<string, unknown>> | undefined
          if (content) {
            for (const blk of content) {
              if (blk.type === "text" && typeof blk.text === "string") {
                buffer += blk.text
                fullText += blk.text
                await flushBuffer()
              }
            }
          }
        }

        if (msgType === "result") {
          const msg = message as Record<string, unknown>
          const usage = msg.usage as Record<string, unknown> | undefined
          if (usage) {
            inputTokens = usage.input_tokens as number | undefined
            outputTokens = usage.output_tokens as number | undefined
          }
          costUsd = msg.total_cost_usd as number | undefined
          if (resolvedModel) {
            await ctx.runMutation(internal.generations.setClaudeResolvedModel, {
              sessionId: args.sessionId,
              model: resolvedModel,
            })
          }
        }
      }

      await flushBuffer()

      const gen = await ctx.runQuery(internal.generations.getInternal, {
        generationId: args.generationId,
      })
      if (gen?.status === "cancelled") {
        await ctx.runMutation(internal.research.resetPlaceholderBlock, { blockId: args.blockId })
        trace.complete({ text: fullText, inputTokens, outputTokens, costUsd, durationMs: Date.now() - startTime, resolvedModel })
        await flushLangfuse()
        return
      }

      // Fill the placeholder block with the research result
      await ctx.runMutation(internal.research.fillPlaceholderBlock, {
        blockId: args.blockId,
        content: fullText,
      })

      await ctx.runMutation(internal.generations.completeWithUsage, {
        generationId: args.generationId,
        inputTokens,
        outputTokens,
        costUsd,
        durationMs: Date.now() - startTime,
      })

      trace.complete({ text: fullText, inputTokens, outputTokens, costUsd, durationMs: Date.now() - startTime, resolvedModel })
      await flushLangfuse()
    } catch (error) {
      if (error instanceof Error && error.name === "AbortError") {
        await flushBuffer()
        await ctx.runMutation(internal.research.resetPlaceholderBlock, { blockId: args.blockId })
        trace.complete({ text: fullText, inputTokens, outputTokens, costUsd, durationMs: Date.now() - startTime, resolvedModel })
        await flushLangfuse()
        return
      }

      const errorMessage = error instanceof Error ? error.message : String(error)
      const stderrOutput = stderrChunks.join("").trim()
      await flushBuffer()
      await ctx.runMutation(internal.research.resetPlaceholderBlock, { blockId: args.blockId })
      const fullError = stderrOutput
        ? `Research error: ${errorMessage}\nstderr: ${stderrOutput}`
        : `Research error: ${errorMessage}`
      await ctx.runMutation(internal.generations.fail, {
        generationId: args.generationId,
        error: fullError,
      })
      trace.error(fullError)
      await flushLangfuse()
    }
  },
})
```

- [ ] **Step 1: Create PlaceholderBlock.tsx**

```typescript
import { useState } from "react"
import { useQuery, useMutation } from "convex/react"
import { api } from "../../convex/_generated/api"
import type { Id } from "../../convex/_generated/dataModel"
import { Button } from "@/components/ui/button"

interface PlaceholderBlockProps {
  blockId: Id<"blocks">
  sessionId: Id<"sessions">
  placeholderSpec?: string
  placeholderStatus?: "empty" | "running" | "filled"
  content: string
}

export function PlaceholderBlock({
  blockId,
  sessionId,
  placeholderSpec,
  placeholderStatus = "empty",
  content,
}: PlaceholderBlockProps) {
  const [spec, setSpec] = useState(placeholderSpec ?? "")
  const [isSaving, setIsSaving] = useState(false)
  const [isStarting, setIsStarting] = useState(false)
  const [error, setError] = useState<string | null>(null)

  const updateBlock = useMutation(api.blocks.update)
  const startResearch = useMutation(api.research.startResearch)
  const cancelGeneration = useMutation(api.generations.cancel)

  // Subscribe to the latest generation for this session to show streaming progress
  const latestGeneration = useQuery(
    api.generations.getLatestForSession,
    placeholderStatus === "running" ? { sessionId } : "skip"
  )

  const handleSaveSpec = async () => {
    if (!spec.trim()) return
    setIsSaving(true)
    setError(null)
    try {
      await updateBlock({ id: blockId, placeholderSpec: spec.trim() })
    } catch (e) {
      setError(e instanceof Error ? e.message : "Failed to save spec")
    } finally {
      setIsSaving(false)
    }
  }

  const handleRunResearch = async () => {
    setIsStarting(true)
    setError(null)
    try {
      await startResearch({ sessionId })
    } catch (e) {
      setError(e instanceof Error ? e.message : "Failed to start research")
    } finally {
      setIsStarting(false)
    }
  }

  const handleCancel = async () => {
    if (latestGeneration?._id) {
      await cancelGeneration({ generationId: latestGeneration._id })
    }
  }

  if (placeholderStatus === "filled") {
    return (
      <div className="space-y-2">
        <div className="prose prose-sm max-w-none text-sm whitespace-pre-wrap">
          {content}
        </div>
        <details className="text-xs text-muted-foreground">
          <summary className="cursor-pointer">Research spec</summary>
          <p className="mt-1 whitespace-pre-wrap">{placeholderSpec}</p>
        </details>
        <Button
          variant="outline"
          size="sm"
          className="h-7 text-xs"
          onClick={handleRunResearch}
          disabled={isStarting}
        >
          Re-run Research
        </Button>
        {error && <p className="text-xs text-destructive">{error}</p>}
      </div>
    )
  }

  if (placeholderStatus === "running") {
    const streamingText = latestGeneration?.status === "streaming"
      ? latestGeneration.text
      : ""
    return (
      <div className="space-y-2">
        <p className="text-xs text-muted-foreground">Researching...</p>
        {streamingText && (
          <div className="prose prose-sm max-w-none text-sm whitespace-pre-wrap opacity-70">
            {streamingText}
          </div>
        )}
        <Button
          variant="outline"
          size="sm"
          className="h-7 text-xs"
          onClick={handleCancel}
        >
          Cancel
        </Button>
      </div>
    )
  }

  // empty state
  return (
    <div className="space-y-2">
      <textarea
        className="w-full min-h-24 text-sm border rounded p-2 resize-y bg-background"
        placeholder="Describe what to research and what format the output should take..."
        value={spec}
        onChange={(e) => setSpec(e.target.value)}
      />
      <div className="flex gap-2">
        <Button
          variant="outline"
          size="sm"
          className="h-7 text-xs"
          onClick={handleSaveSpec}
          disabled={isSaving || !spec.trim()}
        >
          {isSaving ? "Saving..." : "Save Spec"}
        </Button>
        {placeholderSpec && (
          <Button
            size="sm"
            className="h-7 text-xs"
            onClick={handleRunResearch}
            disabled={isStarting}
          >
            {isStarting ? "Starting..." : "Run Research"}
          </Button>
        )}
      </div>
      {error && <p className="text-xs text-destructive">{error}</p>}
    </div>
  )
}
```
